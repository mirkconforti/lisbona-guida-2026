        // --- TAB SWITCHING LOGIC ---
        function switchTab(tabId) {
            document.querySelectorAll('.tab-content').forEach(el => el.classList.add('hidden'));
            document.getElementById(`tab-${tabId}`).classList.remove('hidden');

            // Update Navigation UI States
            document.querySelectorAll('.nav-btn').forEach(btn => {
                btn.classList.remove('text-brand-cyan', 'bg-slate-800/80', 'border', 'border-brand-cyan/30');
                btn.classList.add('text-slate-400');
            });

            const activeBtn = document.getElementById(`nav-${tabId}`);
            if (activeBtn) {
                activeBtn.classList.remove('text-slate-400');
                activeBtn.classList.add('text-brand-cyan', 'bg-slate-800/80', 'border', 'border-brand-cyan/30');
            }

            // Mobile Nav UI update
            document.querySelectorAll('.mob-btn').forEach(btn => btn.classList.remove('text-brand-cyan'));
            const activeMobBtn = document.getElementById(`mob-nav-${tabId}`);
            if (activeMobBtn) activeMobBtn.classList.add('text-brand-cyan');

            // Initialize Map if opened for the first time
            if (tabId === 'map' && !leafletMap) {
                setTimeout(initMap, 200);
            }
        }

        // --- MAP LOGIC (LEAFLET) ---
        function initMap() {
            leafletMap = L.map('map-container').setView([38.7223, -9.1393], 12);
            
            // Dark Mode Tile Layer (CartoDB Dark Matter)
            L.tileLayer('https://{s}.basemaps.cartocdn.com/dark_all/{z}/{x}/{y}{r}.png', {
                attribution: '&copy; OpenStreetMap &copy; CARTO',
                maxZoom: 19
            }).addTo(leafletMap);

            markersGroup = L.layerGroup().addTo(leafletMap);
            renderMapMarkers('all');
        }

        function renderMapMarkers(filter) {
            markersGroup.clearLayers();

            mapMarkers.forEach(m => {
                if (filter === 'all' || m.type === filter) {
                    const marker = L.circleMarker([m.lat, m.lng], {
                        radius: 8,
                        fillColor: m.type === 'food' ? '#f59e0b' : '#00f2fe',
                        color: '#ffffff',
                        weight: 2,
                        opacity: 1,
                        fillOpacity: 0.8
                    });

                    marker.bindPopup(`
                        <div style="font-family: 'Plus Jakarta Sans', sans-serif;">
                            <strong style="color: #00f2fe; font-size: 13px;">${m.title}</strong>
                            <p style="margin-top: 4px; font-size: 11px; color: #cbd5e1;">${m.desc}</p>
                        </div>
                    `);

                    markersGroup.addLayer(marker);
                }
            });
        }

        function filterMapMarkers(type) {
            renderMapMarkers(type);
        }

        function speakPortuguese(text) {
            if ('speechSynthesis' in window) {
                const utterance = new SpeechSynthesisUtterance(text);
                utterance.lang = 'pt-PT';
                utterance.rate = 0.9;
                window.speechSynthesis.speak(utterance);
            } else {
                alert("Sintesi vocale non supportata dal tuo browser.");
            }
        }

        // --- BUDGET LOGIC ---
        function addExpense() {
            const title = document.getElementById('expense-title').value.trim();
            const amount = parseFloat(document.getElementById('expense-amount').value);
            const category = document.getElementById('expense-category').value;

            if (!title || isNaN(amount) || amount <= 0) {
                alert("Inserisci un titolo e un importo valido.");
                return;
            }

            expensesList.push({ title, amount, category, date: new Date().toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'}) });
            
            // Clear inputs
            document.getElementById('expense-title').value = '';
            document.getElementById('expense-amount').value = '';

            updateBudgetUI();
        }

        function updateBudgetUI() {
            const totalSpent = expensesList.reduce((acc, curr) => acc + curr.amount, 0);
            const totalBudget = 900.00;
            const remaining = totalBudget - totalSpent;

            document.getElementById('total-spent-display').innerText = `€ ${totalSpent.toFixed(2)}`;
            document.getElementById('total-remaining-display').innerText = `€ ${remaining.toFixed(2)}`;

            const listContainer = document.getElementById('expense-list');
            listContainer.innerHTML = '';

            if (expensesList.length === 0) {
                listContainer.innerHTML = `<p class="text-xs text-slate-500 text-center py-4">Nessuna spesa inserita finora.</p>`;
                return;
            }

            expensesList.forEach((exp, index) => {
                const item = document.createElement('div');
                item.className = "flex items-center justify-between p-3 rounded-xl bg-slate-900/60 border border-slate-800 text-xs";
                item.innerHTML = `
                    <div>
                        <p class="font-bold text-white">${exp.title}</p>
                        <p class="text-[10px] text-slate-400">${exp.category} • ${exp.date}</p>
                    </div>
                    <span class="font-bold text-brand-purple">€ ${exp.amount.toFixed(2)}</span>
                `;
                listContainer.appendChild(item);
            });
        }

            });
        }

        // --- COUNTDOWN TIMER LOGIC ---
        function startCountdown() {
            const targetDate = new Date("September 3, 2026 13:00:00").getTime();

            setInterval(() => {
                const now = new Date().getTime();
                const distance = targetDate - now;

                if (distance < 0) {
                    document.getElementById("countdown-timer").innerText = "IN VIAGGIO!";
                    return;
                }

                const days = Math.floor(distance / (1000 * 60 * 60 * 24));
                const hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
                const minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));

                document.getElementById("countdown-timer").innerText = `${days}d ${hours}h ${minutes}m`;
            }, 1000);
        }

            const inputEl = document.getElementById('chat-input');
            const query = inputEl.value.trim();
            if (!query) return;

            const messagesContainer = document.getElementById('chat-messages');

            // Append User Message
            const userMsgDiv = document.createElement('div');
            userMsgDiv.className = "flex justify-end space-x-2";
            userMsgDiv.innerHTML = `

                    ${query}
                </div>
            `;
            messagesContainer.appendChild(userMsgDiv);
            inputEl.value = '';

            // Scroll Bottom
            messagesContainer.scrollTop = messagesContainer.scrollHeight;

            const loadingDiv = document.createElement('div');
            loadingDiv.className = "flex space-x-3 loading-ai";
            loadingDiv.innerHTML = `
                <div class="w-8 h-8 rounded-full bg-brand-purple/20 border border-brand-purple/50 flex items-center justify-center flex-shrink-0">

                </div>
            `;
            messagesContainer.appendChild(loadingDiv);
            messagesContainer.scrollTop = messagesContainer.scrollHeight;
            lucide.createIcons();

            // Real API Call to Gemini Endpoint
            const apiKey = "";
            const systemPrompt = "Sei Horizon AI, la guida turistica virtuale ed esperta di Lisbona per due viaggiatori (1 adulto e 1 ragazzo di 17 anni) alloggiati all'hotel ibis Lisboa Saldanha dal 3 al 9 settembre 2026. Rispondi in modo conciso, pratico e amichevole in italiano.";

            try {

                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        contents: [{ parts: [{ text: query }] }],
                        systemInstruction: { parts: [{ text: systemPrompt }] }
                    })
                });

                const data = await response.json();
                const replyText = data.candidates?.[0]?.content?.parts?.[0]?.text || "Scusami, ho incontrato un piccolo errore di rete. Posso comunque aiutarti a navigare le schede dell'itinerario!";

            }
